% Define the default multiplier
\newcommand{\multiplier}{4}

% Define the macro
\newcommand{\symbolscale}[1]{%
    \pgfmathsetmacro{\value}{round(#1 * \multiplier)}% Multiply by the multiplier and round to nearest integer
    \pgfmathtruncatemacro{\valueint}{\value}% Truncate to ensure an integer
    % Initialize the counter loop
    \newcount\counter
    \counter=1
    \loop
        \ifnum\counter>\valueint
            $\circ$%
        \else
            $\bullet$%
        \fi
        \advance\counter by 1 % Increment counter
    \ifnum\counter<\numexpr\multiplier+1\relax
    \repeat
}
